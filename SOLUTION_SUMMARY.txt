🚀 CÓMO USAR
═════════════════════════════════════════════════════════════════════════════════

  1. Mostrar nombre de módulo:
     ────────────────────────────
     import { getDisplayName } from "@/lib/modules-display";
     
     <h1>{getDisplayName("presupuesto-cdp")}</h1>
     // Output: "CDP" ✨

  2. Mostrar label de item de navegación:
     ────────────────────────────────────
     import { getItemLabel } from "@/lib/navigation";
     
     <button>{getItemLabel(item)}</button>
     // Output: "Alertas Presupuestales" ✨

  3. Validar en desarrollo:
     ──────────────────────
     validateDisplayText(text); // Avisa si es técnico

  4. Buscar módulos:
     ────────────────
     searchModules("presupuesto")
     // Output: [["presupuesto-cdp", "CDP"], ...]
